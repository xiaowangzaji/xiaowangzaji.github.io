3.变量与常量

// 变量声明
int count;          // 未初始化
double price = 19.99; // 初始化

// 常量定义
const int MAX_SIZE = 100;     // 编译时常量
#define PI 3.1415926          // 宏定义（不推荐）
constexpr int ARRAY_SIZE = 50; // C++11 常量表达式

// 类型推导（C++11）
auto num = 10;        // 自动推导为int
auto name = "Alice";  // 推导为const char*

4.运算符

算术运算符

int a = 10, b = 3;
a + b;   // 13
a - b;   // 7
a * b;   // 30
a / b;   // 3 (整数除法)
a % b;   // 1
a++;     // 后置递增
--b;     // 前置递减

关系运算符

==  // 等于
!=  // 不等于

>   // 大于
>   <   // 小于
>   =  // 大于等于
>   <=  // 小于等于

逻辑运算符

&&  // 逻辑与
||  // 逻辑或
!   // 逻辑非

5. 输入输出

#include <iostream>
using namespace std;

int main() {
    int number;
    
    // 标准输入
    cout << "Enter a number: ";
    cin >> number;

    // 格式化输出
    cout << "You entered: " << number << endl;
    return 0;
}
